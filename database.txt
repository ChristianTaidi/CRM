CREATE TABLE COSTUMER (
ID_COSTUMER INT (5) NOT NULL PRIMARY KEY,
NAME_COSTUMER CHAR (30),
TYPE_COSTUMER CHAR (25),
CITY CHAR (30),
COUNTRY CHAR (30));

CREATE TABLE DESIGNER (
ID_DESIGNER INT(5) NOT NULL PRIMARY KEY,
NAME_DESIGNER VARCHAR (30) NOT NULL,
JOIN_DATE DATE NOT NULL DEFAULT CURDATE,
NUMBER_OF_DESIGNS INT (5) NOT NULL DEFAULT 0,
CITY CHAR (30) NOT NULL,
COUNTRY CHAR (30) NOT NULL);

CREATE TABLE DESIGNERCOSTUMER(
ID_DESIGN INT (5) NOT NULL PRIMARY KEY,
ID_COSTUMER INT (5) FOREIGN KEY REFERENCES (COSTUMER),
ID_DESIGNER INT (5) FOREIGN KEY REFERENCES (DESIGNER)
ID_DESIGN INT (5) FOREIGN KEY REFERENCES (DESIGN));

CREATE TABLE DESIGNE(
ID_DESIGNE INT (5) NOT NULL PRIMARY KEY,
ID_DESIGNER INT (5) FOREIGN KEY REFERENCES (DESIGNER),
VALUE INT (10) NOT NULL DEFAULT 0,
START_DATE DATE NOT NULL DEFAULT CURDATE,
FINISH_DATE DATE,
DESCRIPTION VARCHAR (255));


CREATE TABLE CAMPAIGN (
ID_CAMPAIGN INT(5) NOT NULL PRIMARY KEY,
ID_DESIGNER INT (5) FOREIGN KEY REFERENCES (DESIGNER),
TYPE_CAMPAIGN CHAR (30),
OBJECTIVE_COSTUMER CHAR (25),
DESCRIPTION VARCHAR (255),
START_DATE DATE DEFAULT CURDATE,
FINISH_DATE DATE,
VALUE INT (10) DEFAULT 0);
