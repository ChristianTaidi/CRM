DROP table CUSTOMER;
DROP TABLE DESIGNER;
DROP TABLE DESIGN;
DROP TABLE DESIGN_ORDER;
DROP TABLE CAMPAIGN;

CREATE TABLE CUSTOMER (
ID_CUSTOMER integer primary key autoincrement,
NAME_CUSTOMER VARCHAR (30),
TYPE_CUSTOMER VARCHAR (25),
CITY VARCHAR (30),
COUNTRY VARCHAR (30),
CUSTOMER_AGE int,
BUDGET int);

CREATE TABLE DESIGNER (
ID_DESIGNER integer primary key autoincrement,
NAME_DESIGNER text (30) NOT NULL,
JOIN_DATE DATE NOT NULL DEFAULT CURDATE,
NUMBER_OF_DESIGNS integer (5) NOT NULL DEFAULT 0,
CITY text (30) NOT NULL,
COUNTRY text (30) NOT NULL);

CREATE TABLE DESIGN(
ID_DESIGN integer primary key autoincrement,
ID_DESIGNER integer,
VALUE integer (10) NOT NULL DEFAULT 0,
START_DATE DATE NOT NULL DEFAULT CURRENT_DATE,
FINISH_DATE DATE,
DESCRIPTION text (255),
foreign key(ID_DESIGNER) references DESIGNER(ID_DESIGNER));

CREATE TABLE DESIGN_ORDER(
ID_ORDER integer primary key autoincrement,
ID_DESIGN integer,
ID_CUSTOMER integer,
foreign key(ID_CUSTOMER) references CUSTOMER(ID_CUSTOMER),
foreign key(ID_DESIGN) references DESIGN(ID_DESIGN));



CREATE TABLE CAMPAIGN (
ID_CAMPAIGN integer primary key autoincrement,
TYPE_CAMPAIGN text (30),
OBJECTIVE_CUSTOMER text (25),
DESCRIPTION text (255),
START_DATE DATE DEFAULT CURRENT_DATE,
FINISH_DATE DATE,
VALUE integer (10) DEFAULT 0);
